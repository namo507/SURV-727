find_census_table <- function(tables) {
  Filter(Negate(is.null), lapply(tables, function(tbl) {
    if (any(grepl("Census", 
                  gsub("\\s+", "", colnames(tbl)), 
                  ignore.case = TRUE))) {
      return(tbl)
    } else {
      return(NULL)
    }
  }))
}

df_cleaned <- function(ltbl){
  n <- length(ltbl)
  df <- list()
  for(i in 1:n){
    df[[i]] <- ltbl[[i]][,-3]
    df[[i]] <- df[[i]][-nrow(df[[i]]),]
  }
